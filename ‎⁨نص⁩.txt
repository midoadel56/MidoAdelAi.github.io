    <!-- سكربت التشغيل (JavaScript) -->
    <script>
        // --- إعدادات الـ API ---
        // سيتم حقن المفتاح تلقائياً في بيئة العرض الحالية.
        // عند رفع الموقع خارجاً، قم بوضع مفتاحك الخاص بين علامتي التنصيص.
        const apiKey = ""; 

        // عناصر واجهة المستخدم
        const promptInput = document.getElementById('promptInput');
        const generateBtn = document.getElementById('generateBtn');
        const resultArea = document.getElementById('resultArea');
        const loadingIndicator = document.getElementById('loadingIndicator');
        const imageContainer = document.getElementById('imageContainer');
        const generatedImage = document.getElementById('generatedImage');
        const errorMessage = document.getElementById('errorMessage');

        let currentImageBase64 = "";

        async function generateImage() {
            const prompt = promptInput.value.trim();
            
            if (!prompt) {
                alert("يرجى إدخال وصف للصورة أولاً!");
                return;
            }

            // إعداد واجهة المستخدم لحالة التحميل
            resultArea.classList.remove('hidden');
            loadingIndicator.classList.remove('hidden');
            loadingIndicator.classList.add('flex');
            imageContainer.classList.add('hidden');
            errorMessage.classList.add('hidden');
            generateBtn.disabled = true;
            generateBtn.classList.add('opacity-50', 'cursor-not-allowed');

            try {
                // استدعاء نموذج Nano Banana (gemini-2.5-flash-image-preview)
                const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-image-preview:generateContent?key=${apiKey}`;
                
                const payload = {
                    contents: [
                        {
                            parts: [{ text: prompt }]
                        }
                    ],
                    generationConfig: {
                        responseModalities: ["IMAGE"]
                    }
                };

                const response = await fetch(url, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(payload)
                });

                if (!response.ok) {
                    throw new Error(`HTTP error! status: ${response.status}`);
                }

                const data = await response.json();
                
                // استخراج الصورة بصيغة Base64 من الرد
                const parts = data.candidates?.[0]?.content?.parts;
                const imagePart = parts?.find(p => p.inlineData);
                
                if (imagePart && imagePart.inlineData && imagePart.inlineData.data) {
                    currentImageBase64 = imagePart.inlineData.data;
                    const mimeType = imagePart.inlineData.mimeType || 'image/png';
                    
                    // عرض الصورة
                    generatedImage.src = `data:${mimeType};base64,${currentImageBase64}`;
                    
                    loadingIndicator.classList.add('hidden');
                    loadingIndicator.classList.remove('flex');
                    imageContainer.classList.remove('hidden');
                } else {
                    throw new Error("لم يتم العثور على صورة في الرد.");
                }

            } catch (error) {
                console.error("Error generating image:", error);
                loadingIndicator.classList.add('hidden');
                loadingIndicator.classList.remove('flex');
                errorMessage.classList.remove('hidden');
            } finally {
                // إعادة تفعيل الزر
                generateBtn.disabled = false;
                generateBtn.classList.remove('opacity-50', 'cursor-not-allowed');
            }
        }

        // دالة لتحميل الصورة لجهاز المستخدم
        function downloadImage() {
            if (!currentImageBase64) return;
            
            const link = document.createElement('a');
            link.href = generatedImage.src;
            link.download = `generated-image-${Date.now()}.png`;
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }
    </script>
